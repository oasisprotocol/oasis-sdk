1use tendermint::{crypto::signature, PublicKey, Signature};

3use crate::{common::crypto::hash::Hash, consensus::tendermint::TENDERMINT_CONTEXT};

5/// A signature verifier that uses the Oasis Core domain separation scheme.
6#[derive(Copy, Clone, Debug, Default, PartialEq, Eq)]
7pub struct DomSepVerifier;
8
9impl signature::Verifier for DomSepVerifier {
10    fn verify(
11        pubkey: PublicKey,
12        msg: &[u8],
13        signature: &Signature,
14    ) -> Result<(), signature::Error> {
15        // Use Oasis Core domain separation scheme.
16        let msg = Hash::digest_bytes_list(&[TENDERMINT_CONTEXT, msg]);
17
18        // Forward the actual verification to the Tendermint's default verifier.
19        tendermint::crypto::default::signature::Verifier::verify(pubkey, msg.as_ref(), signature)
20    }
21}
