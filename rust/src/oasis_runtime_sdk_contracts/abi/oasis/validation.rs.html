15fn check_valtype_acceptable(ty: ValType) -> Result<(), Error> {
16    match ty {
17        ValType::F32 | ValType::F64 => Err(Error::ModuleUsesFloatingPoint),
18        _ => Ok(()),
19    }
20}

411impl<Cfg: Config> OasisV1<Cfg> {
412    pub(super) fn validate_module(
413        &self,
414        module: &Module,
415        params: &Parameters,
416    ) -> Result<Info, Error> {
417        // Verify that all required exports are there.
418        let exports: BTreeSet<&str> = module
419            .exports
420            .iter()
421            .map(|export| export.name.as_str())
422            .collect();
423        for required in Self::REQUIRED_EXPORTS {
424            if !exports.contains(required) {
425                return Err(Error::CodeMissingRequiredExport(required.to_string()));
426            }
427        }
428
429        for reserved in Self::RESERVED_EXPORTS {
430            if exports.contains(reserved) {
431                return Err(Error::CodeDeclaresReservedExport(reserved.to_string()));
432            }
433        }
434
435        // Determine supported ABI sub-version.
436        let sv_exports: Vec<_> = exports
437            .iter()
438            .filter(|export| export.starts_with(EXPORT_SUB_VERSION_PREFIX))
439            .collect();
440        let abi_sv = match sv_exports[..] {
441            [] => {
442                // No versions, this is v0.
443                0
444            }
445            [sv] => {
446                // A single version, parse which one.
447                sv.strip_prefix(EXPORT_SUB_VERSION_PREFIX)
448                    .ok_or(Error::CodeMalformed)?
449                    .parse::<u32>()
450                    .map_err(|_| Error::CodeMalformed)?
451            }
452            _ => {
453                // Multiple versions.
454                return Err(Error::CodeDeclaresMultipleSubVersions);
455            }
456        };
457
458        // Verify that there is no start function defined.
459        if module.start.is_some() {
460            return Err(Error::CodeDeclaresStartFunction);
461        }
462
463        // Verify that there is at most one memory defined.
464        if module.memories.iter().count() > 1 {
465            return Err(Error::CodeDeclaresTooManyMemories);
466        }
467
468        // Verify that the code doesn't use any floating point instructions.
469        let mut function_count = 0u32;
470        for func in module.functions() {
471            let func_type = module.types.get(func.ty());
472            for val_type in func_type.params().iter().chain(func_type.results().iter()) {
473                check_valtype_acceptable(*val_type)?;
474            }
475            if let walrus::FunctionKind::Local(local) = &func.kind {
476                function_count += 1;
477                if function_count > params.max_wasm_functions {
478                    return Err(Error::CodeDeclaresTooManyFunctions);
479                }
480                let mut scanner = FloatScanner(false);
481                dfs_in_order(&mut scanner, local, local.entry_block());
482                if scanner.0 {
483                    return Err(Error::ModuleUsesFloatingPoint);
484                }
485            }
486        }
487
488        // ... or tables with floating point elements.
489        for table in module.tables.iter() {
490            check_valtype_acceptable(table.element_ty)?;
491        }
492
493        // ... or floating point elements.
494        for element in module.elements.iter() {
495            check_valtype_acceptable(element.ty)?;
496        }
497
498        // ... or floating point globals.
499        for global in module.globals.iter() {
500            check_valtype_acceptable(global.ty)?;
501        }
502
503        // ... just don't think about floats in any way.
504        let mut local_count = 0u32;
505        for local in module.locals.iter() {
506            local_count += 1;
507            if local_count > params.max_wasm_locals {
508                return Err(Error::CodeDeclaresTooManyLocals);
509            }
510            check_valtype_acceptable(local.ty())?;
511        }
512
513        Ok(Info { abi_sv })
514    }
