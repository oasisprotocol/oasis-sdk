50    fn raw_call_with_request_context<'ctx, C: Context>(
51        ctx: &mut ExecutionContext<'ctx, C>,
52        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
53        request: &[u8],
54        deposited_tokens: &[token::BaseUnits],
55        function_name: &str,
56    ) -> Result<contract_sdk::ExecutionOk, Error> {
57        // Allocate memory for context and request, copy serialized data into the region.
58        let mut ec = contract_sdk::ExecutionContext {
59            instance_id: ctx.instance_info.id,
60            instance_address: ctx.instance_info.address().into(),
61            caller_address: ctx.caller_address.into(),
62            deposited_tokens: deposited_tokens.iter().map(|b| b.into()).collect(),
63            ..Default::default()
64        };
65        if ctx.code_info.abi_sv >= 1 {
66            // Supports read only and call format flags.
67            ec.read_only = ctx.read_only;
68            ec.call_format = ctx.call_format.into();
69        }
70        let context_dst = Self::serialize_and_allocate(instance, ec)
71            .map_err(|err| Error::ExecutionFailed(err.into()))?;
72        let request_dst = Self::allocate_and_copy(instance, request)
73            .map_err(|err| Error::ExecutionFailed(err.into()))?;
74
75        // Call the corresponding function in the smart contract.
76        let result = {
77            // The high-level function signature of the WASM export is as follows:
78            //
79            //   fn(ctx: &contract_sdk::ExecutionContext, request: &[u8]) -> contract_sdk::ExecutionResult
80            //
81            let func = instance
82                .find_function::<((u32, u32), (u32, u32)), u32>(function_name)
83                .map_err(|err| Error::ExecutionFailed(err.into()))?;
84            let result = func
85                .call_with_context(ctx, (context_dst.to_arg(), request_dst.to_arg()))
86                .map_err(|err| Error::ExecutionFailed(err.into()))?;
87            instance
88                .runtime()
89                .try_with_memory(|memory| -> Result<_, Error> {
90                    memory::Region::deref(&memory, result)
91                        .map_err(|err| Error::ExecutionFailed(err.into()))
92                })
93                .unwrap()?
94        };
95
96        // Enforce maximum result size limit before attempting to deserialize it.
97        if result.length as u32 > ctx.params.max_result_size_bytes {
98            return Err(Error::ResultTooLarge(
99                result.length as u32,
100                ctx.params.max_result_size_bytes,
101            ));
102        }
103
104        // Deserialize region into result structure.
105        let result: contract_sdk::ExecutionResult = instance
106            .runtime()
107            .try_with_memory(|memory| -> Result<_, Error> {
108                let data = result
109                    .as_slice(&memory)
110                    .map_err(|err| Error::ExecutionFailed(err.into()))?;
111
112                cbor::from_slice(data).map_err(|err| Error::ExecutionFailed(err.into()))
113            })
114            .unwrap()?;
115
116        match result {
117            contract_sdk::ExecutionResult::Ok(ok) => Ok(ok),
118            contract_sdk::ExecutionResult::Failed {
119                module,
120                code,
121                message,
122            } => Err(ContractError::new(ctx.instance_info.code_id, &module, code, &message).into()),
123        }
124    }

126    fn call_with_request_context<'ctx, C: Context>(
127        ctx: &mut ExecutionContext<'ctx, C>,
128        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
129        request: &[u8],
130        deposited_tokens: &[token::BaseUnits],
131        function_name: &str,
132    ) -> ExecutionResult {
133        // Fetch initial gas counter value so we can determine how much gas was used.
134        let initial_gas = gas::get_remaining_gas(instance);
135
136        let inner = Self::raw_call_with_request_context(
137            ctx,
138            instance,
139            request,
140            deposited_tokens,
141            function_name,
142        )
143        .map_err(|err| {
144            // Check if an abort flag has been set and propagate the error.
145            if let Some(aborted) = ctx.aborted.take() {
146                return aborted;
147            }
148
149            // Check if call failed due to gas being exhausted and return a proper error.
150            let exhausted_gas = gas::get_exhausted_amount(instance);
151            if exhausted_gas != 0 {
152                // Compute how much gas was wanted.
153                let final_gas = gas::get_remaining_gas(instance);
154                let wanted_gas = initial_gas + exhausted_gas.saturating_sub(final_gas);
155                core::Error::out_of_gas::<<<C::Runtime as Runtime>::Core as core::API>::Config>(
156                    initial_gas,
157                    wanted_gas,
158                )
159                .into()
160            } else {
161                err
162            }
163        });
164
165        // Compute how much gas (in SDK units) was actually used.
166        let final_gas = gas::get_remaining_gas(instance);
167        let gas_used = initial_gas.saturating_sub(final_gas) / GAS_SCALING_FACTOR;
168
169        ExecutionResult { inner, gas_used }
170    }
171}

173impl<Cfg: Config, C: Context> Abi<C> for OasisV1<Cfg> {
174    fn validate(&self, module: &mut walrus::Module, params: &Parameters) -> Result<Info, Error> {
175        let info = self.validate_module(module, params)?;
176
177        // Add gas metering instrumentation.
178        gas::transform(module);
179
180        Ok(info)
181    }
182
183    fn link(
184        &self,
185        instance: &mut wasm3::Instance<'_, '_, ExecutionContext<'_, C>>,
186    ) -> Result<(), Error> {
187        // Storage imports.
188        Self::link_storage(instance)?;
189        // Environment imports.
190        Self::link_env(instance)?;
191        // Crypto imports.
192        Self::link_crypto(instance)?;
193
194        Ok(())
195    }
196
197    fn set_gas_limit(
198        &self,
199        instance: &mut wasm3::Instance<'_, '_, ExecutionContext<'_, C>>,
200        gas_limit: u64,
201    ) -> Result<(), Error> {
202        // Derive gas limit from remaining transaction gas based on a scaling factor.
203        let gas_limit = gas_limit.saturating_mul(GAS_SCALING_FACTOR);
204        gas::set_gas_limit(instance, gas_limit)?;
205
206        Ok(())
207    }
208
209    fn instantiate<'ctx>(
210        &self,

213        request: &[u8],
214        deposited_tokens: &[token::BaseUnits],
215    ) -> ExecutionResult {

220            deposited_tokens,
