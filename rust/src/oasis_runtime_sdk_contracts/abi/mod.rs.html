13/// Trait for any WASM ABI to implement.
14pub trait Abi<C: Context> {
15    /// Validate that the given WASM module conforms to the ABI.
16    fn validate(&self, module: &mut walrus::Module, params: &Parameters) -> Result<Info, Error>;
17
18    /// Link required functions into the WASM module instance.
19    fn link(
20        &self,
21        instance: &mut wasm3::Instance<'_, '_, ExecutionContext<'_, C>>,
22    ) -> Result<(), Error>;
23
24    /// Set the gas limit for any following executions.
25    ///
26    /// The specified gas limit should be in regular SDK gas units, not in WASM gas units. The ABI
27    /// should perform any necessary conversions if required.
28    fn set_gas_limit(
29        &self,
30        instance: &mut wasm3::Instance<'_, '_, ExecutionContext<'_, C>>,
31        gas_limit: u64,
32    ) -> Result<(), Error>;
33
34    /// Instantiate a contract.
35    fn instantiate<'ctx>(
36        &self,
37        ctx: &mut ExecutionContext<'ctx, C>,
38        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
39        request: &[u8],
40        deposited_tokens: &[token::BaseUnits],
41    ) -> ExecutionResult;
42
43    /// Call a contract.
44    fn call<'ctx>(
45        &self,
46        ctx: &mut ExecutionContext<'ctx, C>,
47        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
48        request: &[u8],
49        deposited_tokens: &[token::BaseUnits],
50    ) -> ExecutionResult;
51
52    /// Invoke the contract's reply handler.
53    fn handle_reply<'ctx>(
54        &self,
55        ctx: &mut ExecutionContext<'ctx, C>,
56        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
57        reply: Reply,
58    ) -> ExecutionResult;
59
60    /// Invoke the contract's pre-upgrade handler.
61    fn pre_upgrade<'ctx>(
62        &self,
63        ctx: &mut ExecutionContext<'ctx, C>,
64        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
65        request: &[u8],
66        deposited_tokens: &[token::BaseUnits],
67    ) -> ExecutionResult;
68
69    /// Invoke the contract's post-upgrade handler.
70    fn post_upgrade<'ctx>(
71        &self,
72        ctx: &mut ExecutionContext<'ctx, C>,
73        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
74        request: &[u8],
75        deposited_tokens: &[token::BaseUnits],
76    ) -> ExecutionResult;
77
78    /// Query a contract.
79    fn query<'ctx>(
80        &self,
81        ctx: &mut ExecutionContext<'ctx, C>,
82        instance: &wasm3::Instance<'_, '_, ExecutionContext<'ctx, C>>,
83        request: &[u8],
84    ) -> ExecutionResult;
85}

90    pub abi_sv: u32,
91}

93/// Execution context.
94pub struct ExecutionContext<'ctx, C: Context> {
95    /// Transaction context.
96    pub tx_context: &'ctx C,
97    /// Contracts module parameters.
98    pub params: &'ctx Parameters,
99
100    /// Code information.
101    pub code_info: &'ctx types::Code,
102    /// Contract instance information.
103    pub instance_info: &'ctx types::Instance,
104    /// Gas limit for this contract execution.
105    pub gas_limit: u64,
106
107    /// Address of the caller.
108    pub caller_address: Address,
109    /// Whether the call is read-only and must not make any storage modifications.
110    pub read_only: bool,
111    /// Call format.
112    pub call_format: CallFormat,
113
114    /// Whether the execution has aborted with an error that should be propagated instead of just
115    /// using the generic "execution failed" error.
116    pub aborted: Option<Error>,
117}
118
119impl<'ctx, C: Context> ExecutionContext<'ctx, C> {
120    /// Create a new execution context.
121    #[allow(clippy::too_many_arguments)]
122    pub fn new(
123        params: &'ctx Parameters,
124        code_info: &'ctx types::Code,
125        instance_info: &'ctx types::Instance,
126        gas_limit: u64,
127        caller_address: Address,
128        read_only: bool,
129        call_format: CallFormat,
130        tx_context: &'ctx C,
131    ) -> Self {
132        Self {
133            tx_context,
134            params,
135            code_info,
136            instance_info,
137            gas_limit,
138            caller_address,
139            read_only,
140            call_format,
141            aborted: None,
142        }
143    }
144}
