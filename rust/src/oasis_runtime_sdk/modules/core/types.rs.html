18/// Arguments for the EstimateGas query.
19#[derive(Clone, Debug, cbor::Encode, cbor::Decode)]
20#[cbor(no_default)]
21pub struct EstimateGasQuery {
22    /// The address of the caller for which to do estimation. If not specified the authentication
23    /// information from the passed transaction is used.
24    #[cbor(optional)]
25    pub caller: Option<CallerAddress>,
26    /// The unsigned transaction to estimate.
27    pub tx: Transaction,
28    /// If the estimate gas query should fail in case of transaction failures.
29    /// If true, the query will return the transaction error and not the gas estimation.
30    /// Defaults to false.
31    #[cbor(optional)]
32    pub propagate_failures: bool,
33}
