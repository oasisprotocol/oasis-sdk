1use std::sync::Arc;
2
3use anyhow::Result;
4use tokio::sync::mpsc;
5
6use crate::core::common::logger::get_logger;
7
8use super::{processor, App, Environment};
9
10/// Notification to deliver to the application.
11pub(super) enum Notify {
12    RuntimeBlock(u64),
13    RuntimeBlockDone,
14    InitialRegistrationCompleted,
15}
16
17#[derive(Default)]
18struct NotifyState {
19    pending: bool,
20    running: bool,
21}
22
23/// Application notifier task.
24pub(super) struct Task<A: App> {
25    imp: Option<Impl<A>>,
26    tx: mpsc::Sender<Notify>,
27}
28
29impl<A> Task<A>
30where
31    A: App,
32{
33    /// Create an application notifier task.
34    pub(super) fn new(state: Arc<processor::State<A>>, env: Environment<A>) -> Self {
35        let (tx, rx) = mpsc::channel(16);
36
37        let imp = Impl {
38            state,
39            env,
40            logger: get_logger("modules/rofl/app/notifier"),
41            notify: rx,
42            notify_tx: tx.downgrade(),
43        };
44
45        Self { imp: Some(imp), tx }
46    }
47
48    /// Start the application notifier task.
49    pub(super) fn start(&mut self) {
50        if let Some(imp) = self.imp.take() {
51            imp.start();
52        }
53    }
54
55    /// Deliver a notification.
56    pub(super) async fn notify(&self, notification: Notify) -> Result<()> {
57        self.tx.send(notification).await?;
58        Ok(())
59    }
60}
61
62struct Impl<A: App> {
63    state: Arc<processor::State<A>>,
64    env: Environment<A>,
65    logger: slog::Logger,
66
67    notify: mpsc::Receiver<Notify>,
68    notify_tx: mpsc::WeakSender<Notify>,
69}
70
71impl<A> Impl<A>
72where
73    A: App,
74{
75    /// Start the application notifier task.
76    pub(super) fn start(self) {
77        tokio::task::spawn(self.run());
78    }
79
80    /// Run the application notifier task.
81    async fn run(mut self) {
82        slog::info!(self.logger, "starting notifier task");
83
84        // Pending notifications.
85        let mut registered = false;
86        let mut block = NotifyState::default();
87        let mut last_round = 0;
88
89        while let Some(notification) = self.notify.recv().await {
90            match notification {
91                Notify::RuntimeBlock(round) if registered => {
92                    block.pending = true;
93                    last_round = round;
94                }
95                Notify::RuntimeBlock(_) => continue, // Skip blocks before registration.
96                Notify::RuntimeBlockDone => block.running = false,
97                Notify::InitialRegistrationCompleted => registered = true,
98            }
99
100            // Don't do anything unless registered.
101            if !registered {
102                continue;
103            }
104
105            // Block notifications.
106            if block.pending && !block.running {
107                block.pending = false;
108                block.running = true;
109
110                let notify_tx = self.notify_tx.clone();
111                let app = self.state.app.clone();
112                let env = self.env.clone();
113
114                tokio::spawn(async move {
115                    app.on_runtime_block(env, last_round).await;
116                    if let Some(tx) = notify_tx.upgrade() {
117                        let _ = tx.send(Notify::RuntimeBlockDone).await;
118                    }
119                });
120            }
121        }
122
123        slog::info!(self.logger, "notifier task stopped");
124    }
125}
